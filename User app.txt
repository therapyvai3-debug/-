    function verifyAndSubmitTask(taskId, reward, taskTitle, proofType) {
      const proofInput = document.getElementById('proof-' + taskId).value.trim();
      const sBtn = document.getElementById('submit-btn-' + taskId);
      
      if(proofInput.length < 3) { if(typeof tg.showAlert === 'function') tg.showAlert("Error: Please input valid text proof!"); return; }
      if(sBtn) { sBtn.disabled = true; sBtn.innerText = "Processing..."; }
      
      const pushToDatabase = (screenshotData = "N/A") => {
        const submissionRef = db.ref('task_submissions').push();
        submissionRef.set({
          taskId: taskId, taskTitle: taskTitle, userId: currentUser.id.toString(), userName: currentUser.firstName, userUsername: currentUser.username || "N/A",
          proofText: proofInput, screenshotData: screenshotData, reward: Number(reward), status: "PENDING", timestamp: Date.now()
        }).then(() => {
          db.ref('user_completions/' + currentUser.id + '/' + taskId).set({ proof: proofInput, status: "PENDING", submittedAt: Date.now() })
          .then(() => { if(typeof tg.showAlert === 'function') tg.showAlert("আপনার প্রুফ সফলভাবে জমা হয়েছে! 🎉"); switchScreen('tasks'); });
        }).catch((error) => { 
          if(sBtn) { sBtn.disabled = false; sBtn.innerText = "Submit Task"; }
          if(typeof tg.showAlert === 'function') tg.showAlert("ডাটাবেজ ত্রুটি!"); 
        });
      };

      if(proofType === "both") {
        const fileInput = document.getElementById('image-' + taskId);
        if(!fileInput || !fileInput.files || fileInput.files.length === 0) { 
          if(sBtn) { sBtn.disabled = false; sBtn.innerText = "Submit Task"; }
          if(typeof tg.showAlert === 'function') tg.showAlert("Error: Please upload a proof screenshot!"); 
          return; 
        }
        
        // ফাইল অবজেক্টটি ফিক্স করা হয়েছে ([0] ইনডেক্স ব্যবহার করে)
        const file = fileInput.files[0]; 
        const reader = new FileReader();
        reader.onload = function(event) {
          const img = new Image(); 
          img.src = event.target.result;
          img.onload = function() {
            const canvas = document.createElement('canvas'); const ctx = canvas.getContext('2d');
            const MAX_WIDTH = 300; let width = img.width; let height = img.height;
            if (width > MAX_WIDTH) { height *= MAX_WIDTH / width; width = MAX_WIDTH; }
            canvas.width = width; canvas.height = height; ctx.drawImage(img, 0, 0, width, height);
            const compressedBase64 = canvas.toDataURL('image/jpeg', 0.4);
            pushToDatabase(compressedBase64);
          };
          img.onerror = function() {
            if(sBtn) { sBtn.disabled = false; sBtn.innerText = "Submit Task"; }
            if(typeof tg.showAlert === 'function') tg.showAlert("Error loading image file!");
          };
        };
        reader.readAsDataURL(file);
      } else {
        pushToDatabase("N/A");
      }
    }
